def test_add_files_via_dialog(qtbot, test_config, test_db, docx_factory, monkeypatch):
    window = MainWindow(test_config, test_db)
    qtbot.addWidget(window)

    test_files = [docx_factory(filename=f"test{i}.docx") for i in range(2)]

    # Mock file dialog to return test files
    monkeypatch.setattr(
        QFileDialog, 'getOpenFileNames',
        lambda *args, **kwargs: ([str(f) for f in test_files], '')

    assert len(window.selected_files) == 2
    assert window.start_button.isEnabled() is True

def test_start_without_search_term_shows_warning(qtbot, test_config, test_db, docx_factory, monkeypatch):
    window = MainWindow(test_config, test_db)
    qtbot.addWidget(window)

    # Add file but leave search_input empty
    test_file = docx_factory(content="Test", filename="test.docx")
    window.selected_files = [test_file]
    window._update_ui_state()

    # Mock QMessageBox.warning
    warning_calls = []
    monkeypatch.setattr(QMessageBox, 'warning',
        lambda *args: warning_calls.append(args))

    window._on_start_clicked()

    assert len(warning_calls) == 1
    assert "Missing Search Term" in warning_calls[0][1]
